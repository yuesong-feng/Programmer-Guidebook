### 二叉树的层序遍历（非递归实现，使用队列）

```python
def levelOrder(root: TreeNode) -> List[List[int]]:
    if root is None:
        return []
    queen = []
    tree = []
    queen.append(root)
    while len(queen) != 0:
        level = []
        for _ in range(len(queen)):
            p = queen.pop(0)
            level.append(p.val)
            if p.left is not None:
                queen.append(p.left)
            if p.right is not None:
                queen.append(p.right)
        tree.append(level)
    return tree
```
